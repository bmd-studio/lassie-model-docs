Class Event_model

The Event Model handles activities where people can subscribe to.

persons_events_view: DROP VIEW persons_events_view; DROP VIEW persons_def_events_view; CREATE VIEW persons_events_view AS select persons_events.id AS id,persons_events.active AS active,persons_events.event_id AS event_id,persons_events.person_id AS person_id,persons_events.create_timestamp AS create_timestamp,persons_events.source_type_id AS source_type_id,persons_events.transaction_id AS transaction_id,(select persons_def_events.fee AS fee from persons_def_events where (persons_def_events.id = persons_events.event_id)) AS fee from persons_events; CREATE VIEW persons_def_events_view AS SELECT persons_def_events.*, CAST(CONCAT(COUNT(persons_events.id), IF((persons_def_events.max_persons > 0), '/', ''), IF((persons_def_events.max_persons > 0), persons_def_events.max_persons, '')) AS CHAR (255) CHARSET UTF8) AS subscriptions FROM (persons_def_events LEFT JOIN persons_events ON ((persons_events.event_id = persons_def_events.id))) GROUP BY persons_def_events.id;
